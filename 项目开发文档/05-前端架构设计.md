### 3.1 API请求封装

#### request.ts - Axios封装
```typescript
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios'
import { ElMessage } from 'element-plus'
import { useAuthStore } from '@/stores/auth'
import router from '@/router'

// 创建axios实例
const service: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 30000,
  headers: {
    'Content-Type': 'application/json'
  }
})

// 请求拦截器
service.interceptors.request.use(
  (config: AxiosRequestConfig) => {
    const authStore = useAuthStore()
    
    // 添加Token
    if (authStore.token) {
      config.headers.Authorization = `Bearer ${authStore.token}`
    }
    
    return config
  },
  (error) => {
    console.error('请求错误：', error)
    return Promise.reject(error)
  }
)

// 响应拦截器
service.interceptors.response.use(
  (response: AxiosResponse) => {
    const { code, message, data } = response.data
    
    if (code === 200) {
      return data
    } else {
      ElMessage.error(message || '请求失败')
      return Promise.reject(new Error(message))
    }
  },
  (error) => {
    if (error.response) {
      const { status } = error.response
      
      switch (status) {
        case 401:
          ElMessage.error('未授权，请重新登录')
          useAuthStore().logout()
          router.push('/login')
          break
        case 403:
          ElMessage.error('拒绝访问')
          break
        case 404:
          ElMessage.error('请求的资源不存在')
          break
        case 500:
          ElMessage.error('服务器错误')
          break
        default:
          ElMessage.error('网络错误')
      }
    } else {
      ElMessage.error('网络连接失败')
    }
    
    return Promise.reject(error)
  }
)

export default service
```

#### file.ts - 文件接口
```typescript
import request from '@/utils/request'
import type { FileInfo, FileListParams, FileListResponse } from '@/types/file'

export const fileApi = {
  // 获取文件列表
  getFileList(params: FileListParams): Promise<FileListResponse> {
    return request.get('/files', { params })
  },
  
  // 获取文件详情
  getFileDetail(fileId: number): Promise<FileInfo> {
    return request.get(`/files/${fileId}`)
  },
  
  // 上传文件
  uploadFile(formData: FormData, onProgress?: (progress: number) => void) {
    return request.post('/files/upload', formData, {
      headers: { 'Content-Type': 'multipart/form-data' },
      onUploadProgress: (progressEvent) => {
        if (onProgress && progressEvent.total) {
          const percent = Math.round((progressEvent.loaded * 100) / progressEvent.total)
          onProgress(percent)
        }
      }
    })
  },
  
  // 秒传检查
  checkSecUpload(fileMd5: string, fileSize: number, fileName: string) {
    return request.post('/files/sec-upload', { fileMd5, fileSize, fileName })
  },
  
  // 创建文件夹
  createFolder(folderName: string, parentId: number) {
    return request.post('/files/folder', { folderName, parentId })
  },
  
  // 重命名
  renameFile(fileId: number, newName: string) {
    return request.put(`/files/${fileId}/rename`, { newName })
  },
  
  // 删除文件
  deleteFile(fileId: number) {
    return request.delete(`/files/${fileId}`)
  },
  
  // 下载文件
  downloadFile(fileId: number) {
    return request.get(`/files/download/${fileId}`, {
      responseType: 'blob'
    })
  },
  
  // 搜索文件
  searchFiles(keyword: string, pageNum: number, pageSize: number) {
    return request.get('/files/search', {
      params: { keyword, pageNum, pageSize }
    })
  }
}
```

### 3.2 状态管理

#### auth.ts - 认证状态
```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { authApi } from '@/api/auth'
import type { UserInfo } from '@/types/user'
import { storage } from '@/utils/storage'

export const useAuthStore = defineStore('auth', () => {
  const token = ref<string>(storage.get('token') || '')
  const refreshToken = ref<string>(storage.get('refreshToken') || '')
  const userInfo = ref<UserInfo | null>(null)
  
  const isLogin = computed(() => !!token.value)
  const isAdmin = computed(() => userInfo.value?.userType === 1)
  
  // 登录
  const login = async (username: string, password: string) => {
    const data = await authApi.login({ username, password })
    
    token.value = data.accessToken
    refreshToken.value = data.refreshToken
    userInfo.value = data.userInfo
    
    storage.set('token', data.accessToken)
    storage.set('refreshToken', data.refreshToken)
    
    return data
  }
  
  // 退出登录
  const logout = () => {
    token.value = ''
    refreshToken.value = ''
    userInfo.value = null
    
    storage.remove('token')
    storage.remove('refreshToken')
  }
  
  // 刷新Token
  const refresh = async () => {
    const data = await authApi.refreshToken(refreshToken.value)
    token.value = data.accessToken
    storage.set('token', data.accessToken)
  }
  
  return {
    token,
    refreshToken,
    userInfo,
    isLogin,
    isAdmin,
    login,
    logout,
    refresh
  }
})
```

```typescript
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { fileApi } from '@/api/file'
import type { FileInfo } from '@/types/file'

export const useFileStore = defineStore('file', () => {
  const currentFolderId = ref<number>(0)
  const fileList = ref<FileInfo[]>([])
  const selectedFiles = ref<FileInfo[]>([])
  const viewMode = ref<'list' | 'grid' | 'gallery'>('list')
  
  // 加载文件列表
  const loadFileList = async (folderId: number = 0) => {
    currentFolderId.value = folderId
    const data = await fileApi.getFileList({
      parentId: folderId,
      pageNum: 1,
      pageSize: 100
    })
    fileList.value = data.records
  }
  
  // 选择文件
  const selectFile = (file: FileInfo) => {
    const index = selectedFiles.value.findIndex(f => f.id === file.id)
    if (index > -1) {
      selectedFiles.value.splice(index, 1)
    } else {
      selectedFiles.value.push(file)
    }
  }
  
  // 全选
  const selectAll = () => {
    selectedFiles.value = [...fileList.value]
  }
  
  // 清空选择
  const clearSelection = () => {
    selectedFiles.value = []
  }
  
  return {
    currentFolderId,
    fileList,
    selectedFiles,
    viewMode,
    loadFileList,
    selectFile,
    selectAll,
    clearSelection
  }
})
```

#### useUpload.ts - 文件上传逻辑
```typescript
import { ref } from 'vue'
import { ElMessage } from 'element-plus'
import { fileApi } from '@/api/file'
import { calculateFileMd5 } from '@/utils/file'

export function useUpload() {
  const uploadQueue = ref<any[]>([])
  const uploading = ref(false)
  
  // 上传文件
  const uploadFile = async (file: File, parentId: number = 0) => {
    try {
      uploading.value = true
      
      // 计算MD5
      const fileMd5 = await calculateFileMd5(file)
      
      // 检查秒传
      const secUploadResult = await fileApi.checkSecUpload(
        fileMd5,
        file.size,
        file.name
      )
      
      if (secUploadResult.canSecUpload) {
        ElMessage.success('文件已秒传')
        return secUploadResult.fileInfo
      }
      
      // 普通上传
      const formData = new FormData()
      formData.append('file', file)
      formData.append('parentId', String(parentId))
      
      const result = await fileApi.uploadFile(formData, (progress) => {
        console.log('上传进度：', progress)
      })
      
      ElMessage.success('上传成功')
      return result
    } catch (error) {
      ElMessage.error('上传失败')
      throw error
    } finally {
      uploading.value = false
    }
  }
  
  // 批量上传
  const batchUpload = async (files: File[], parentId: number = 0) => {
    for (const file of files) {
      await uploadFile(file, parentId)
    }
  }
  
  return {
    uploadQueue,
    uploading,
    uploadFile,
    batchUpload
  }
}
```

#### file.ts - 文件工具
```typescript
import SparkMD5 from 'spark-md5'

// 计算文件MD5
export const calculateFileMd5 = (file: File): Promise<string> => {
  return new Promise((resolve, reject) => {
    const chunkSize = 2097152 // 2MB
    const chunks = Math.ceil(file.size / chunkSize)
    const spark = new SparkMD5.ArrayBuffer()
    const fileReader = new FileReader()
    let currentChunk = 0
    
    fileReader.onload = (e) => {
      spark.append(e.target?.result as ArrayBuffer)
      currentChunk++
      
      if (currentChunk < chunks) {
        loadNext()
      } else {
        resolve(spark.end())
      }
    }
    
    fileReader.onerror = () => {
      reject(new Error('文件读取失败'))
    }
    
    const loadNext = () => {
      const start = currentChunk * chunkSize
      const end = Math.min(start + chunkSize, file.size)
      fileReader.readAsArrayBuffer(file.slice(start, end))
    }
    
    loadNext()
  })
}

// 格式化文件大小
export const formatFileSize = (bytes: number): string => {
  if (bytes === 0) return '0 B'
  
  const k = 1024
  const sizes = ['B', 'KB', 'MB', 'GB', 'TB']
  const i = Math.floor(Math.log(bytes) / Math.log(k))
  
  return (bytes / Math.pow(k, i)).toFixed(2) + ' ' + sizes[i]
}

// 获取文件扩展名
export const getFileExtension = (fileName: string): string => {
  const lastDot = fileName.lastIndexOf('.')
  return lastDot > -1 ? fileName.substring(lastDot + 1).toLowerCase() : ''
}

// 判断是否为图片
export const isImage = (fileType: string): boolean => {
  return ['jpg', 'jpeg', 'png', 'gif', 'bmp', 'webp', 'svg'].includes(fileType)
}

// 判断是否为视频
export const isVideo = (fileType: string): boolean => {
  return ['mp4', 'avi', 'mov', 'wmv', 'flv', 'mkv'].includes(fileType)
}

// 判断是否为音频
export const isAudio = (fileType: string): boolean => {
  return ['mp3', 'wav', 'flac', 'aac', 'ogg'].includes(fileType)
}
```
